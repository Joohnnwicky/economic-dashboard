   const PCE_SERIES = {
     PCEPI: 'PCEPI',      // Overall PCE inflation
     PCEPILFE: 'PCEPILFE', // Core PCE (ex food/energy)
   };
   
   export async function getPCEData(seriesId: string): Promise<NormalizedIndicator> {
     const data = await fredClient.fetchSeries(seriesId);
     // Override names for PCE-specific display
     if (seriesId === 'PCEPI') {
       data.name = 'PCE物价指数';
     } else if (seriesId === 'PCEPILFE') {
       data.name = '核心PCE物价指数';
     }
     return data;
   }
