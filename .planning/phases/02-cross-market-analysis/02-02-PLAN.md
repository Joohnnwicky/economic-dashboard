   ```typescript
   async function fetchPBOCRate(): Promise<NormalizedIndicator> {
     const response = await fetch('/src/data/pboc-rates.json');
     const data: Array<{date: string, rate: number, type: string}> = await response.json();
     
     // Sort by date descending, latest first
     const sorted = data.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());
     
     const latest = sorted[0];
     const historical = sorted.map(entry => ({
       timestamp: new Date(entry.date),
       value: entry.rate,
     }));
     
     return {
       id: 'pboc-rate',
       name: '中国央行利率 (LPR 1年)',
       value: latest.rate,
       unit: '%',
       timestamp: new Date(latest.date),
       historical,
     };
   }
   
   export function usePBOCRate() {
     return useQuery({
       queryKey: ['pboc-rate'],
       queryFn: fetchPBOCRate,
       staleTime: Infinity,  // Static data, never stale
       gcTime: Infinity,
       retry: false,
     });
   }
   ```
3. Export usePBOCRate

Per RESEARCH.md, PBOC rate is historical static data with rare updates. No API polling needed. staleTime: Infinity prevents unnecessary fetches.

Note: fetch('/src/data/pboc-rates.json') works in Vite dev mode. For production build, may need public/ directory placement or import statement. Test in dev mode first.
