   interface ExportState {
     selectedIndicatorIds: string[];
     exportFormat: 'csv' | 'excel';
     filename: string;
     
     toggleIndicator: (id: string) => void;
     setFormat: (format: 'csv' | 'excel') => void;
     setFilename: (name: string) => void;
     clearSelection: () => void;
   }
   
   export const useExportStore = create<ExportState>((set) => ({
     selectedIndicatorIds: [],
     exportFormat: 'csv',
     filename: '经济指标数据',
     
     toggleIndicator: (id) =>
       set((state) => ({
         selectedIndicatorIds: state.selectedIndicatorIds.includes(id)
           ? state.selectedIndicatorIds.filter((i) => i !== id)
           : [...state.selectedIndicatorIds, id],
       })),
     
     setFormat: (format) => set({ exportFormat: format }),
     setFilename: (name) => set({ filename: name }),
     clearSelection: () => set({ selectedIndicatorIds: [] }),
   }));
   ```

   interface ExportDialogProps {

     availableIndicators: NormalizedIndicator[];

   export function ExportDialog({ isOpen, onClose, availableIndicators }: ExportDialogProps) {
     const { selectedIndicatorIds, exportFormat, filename, toggleIndicator, setFormat, setFilename } = useExportStore();
     
     const selectedIndicators = availableIndicators.filter(ind => selectedIndicatorIds.includes(ind.id));
     
     const handleExport = () => {
       // Sanitize filename (alphanumeric + underscore/hyphen only)
       const sanitizedFilename = filename.replace(/[^\w\-]/g, '_');
       
       if (exportFormat === 'csv') {
         exportToCSV(selectedIndicators, sanitizedFilename);
       } else {
         exportToExcel(selectedIndicators, sanitizedFilename);
       }
       
       onClose();
     };
     
     if (!isOpen) return null;
     
     return (
       <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50">

   interface OverlayComparisonChartProps {
     availableIndicators: NormalizedIndicator[];
     height?: number;
   }
   
   export function OverlayComparisonChart({ availableIndicators, height = 400 }: OverlayComparisonChartProps) {
     const [leftIndicatorId, setLeftIndicatorId] = useState<string | null>(null);
     const [rightIndicatorId, setRightIndicatorId] = useState<string | null>(null);
     
     const leftIndicator = availableIndicators.find(ind => ind.id === leftIndicatorId);
     const rightIndicator = availableIndicators.find(ind => ind.id === rightIndicatorId);
     
     const series = [
       leftIndicator ? { data: leftIndicator, axisPosition: 'left' as const } : null,
       rightIndicator ? { data: rightIndicator, axisPosition: 'right' as const } : null,
     ].filter(Boolean);
     
     if (series.length === 0) {
       return (
         <div className="bg-[#161b22] rounded-lg p-4 flex items-center justify-center h-[${height}px]">
           <p className="text-[#8b949e]">选择指标以进行跨市场对比</p>
         </div>
       );
     }
     
     return (
       <div className="bg-[#161b22] rounded-lg p-4">
         {/* Indicator Selectors */}
         <div className="flex gap-4 mb-4">
           <div className="flex-1">
             <label className="text-[#8b949e] text-sm mb-1 block">左侧指标 (Y轴1)</label>
             <select
               value={leftIndicatorId || ''}
               onChange={(e) => setLeftIndicatorId(e.target.value)}
               className="w-full bg-[#0d1117] text-[#c9d1d9] border border-[#21262d] rounded px-3 py-2"
             >
               <option value="">-- 选择指标 --</option>
               {availableIndicators.map(ind => (
                 <option key={ind.id} value={ind.id}>{ind.name}</option>
               ))}
             </select>
           </div>
           <div className="flex-1">
             <label className="text-[#8b949e] text-sm mb-1 block">右侧指标 (Y轴2)</label>
             <select
               value={rightIndicatorId || ''}
               onChange={(e) => setRightIndicatorId(e.target.value)}
               className="w-full bg-[#0d1117] text-[#c9d1d9] border border-[#21262d] rounded px-3 py-2"
             >
               <option value="">-- 选择指标 --</option>
               {availableIndicators.filter(ind => ind.id !== leftIndicatorId).map(ind => (
                 <option key={ind.id} value={ind.id}>{ind.name}</option>
               ))}
             </select>
           </div>
         </div>
         
         {/* Dual Y-Axis Chart */}
         <MultiSeriesChart series={series} height={height - 80} />
       </div>
     );
   }
   ```

   export function OverlayPanel() {
     // Gather all indicators from hooks
     const fedRate = useFedRate();
     const crypto = useCrypto();
     const employment = useEmploymentSubMetrics();
     const inflation = useInflationSubMetrics();
     const pce = usePCEData();
     const chineseIndices = useChineseIndices();
     const pbocRate = usePBOCRate();
     
     const isLoading = fedRate.isLoading || crypto.isLoading || employment.isLoading || inflation.isLoading;
     
     // Combine all indicators into single array
     const allIndicators: NormalizedIndicator[] = [
       fedRate.data ? [fedRate.data] : [],
       crypto.data || [],
       employment.data || [],
       inflation.data || [],
       pce.data || [],
       chineseIndices.data || [],
       pbocRate.data ? [pbocRate.data] : [],
     ].flat();
     
     if (isLoading) return <LoadingSpinner />;
     if (allIndicators.length === 0) return <ErrorBoundary fallback="数据加载失败，无法进行对比" />;
     
     return (
       <div className="p-4 bg-[#0d1117] rounded-lg">
         <h2 className="text-[#c9d1d9] text-lg mb-4">跨市场对比分析</h2>
         <OverlayComparisonChart availableIndicators={allIndicators} height={450} />
       </div>
     );
