    **src/api/rate-limiter.ts:**
    Implement rate limiting wrapper class:
    ```typescript
    class RateLimiter {
      private lastCallTime: Record<string, number> = {};

      async call<T>(api: string, fn: () => Promise<T>, config: RateLimitConfig): Promise<T> {
        const now = Date.now();
        const elapsed = now - (this.lastCallTime[api] || 0);

        if (elapsed < config.minIntervalMs) {
          await delay(config.minIntervalMs - elapsed);
        }

        this.lastCallTime[api] = Date.now();

        // Log quota estimate
        console.log(`[${api}] Call made, estimated quota remaining...`);

        return fn();
      }
    }
    ```

    Export singleton instance: `export const rateLimiter = new RateLimiter();`

    **src/api/fred.ts:**
    Implement FRED API client:
    ```typescript
    import { rateLimiter } from './rate-limiter';
    import { FRED_BASE_URL, FRED_FED_RATE_SERIES, RATE_LIMITS } from '../constants/api';

    async function getFedRate(timeRange: TimeRange): Promise<NormalizedIndicator> {
      const apiKey = import.meta.env.VITE_FRED_API_KEY;

      if (!apiKey) {
        throw new Error('VITE_FRED_API_KEY not set in .env.local');
      }

      // Calculate date range from timeRange
      const endDate = new Date();
      const startDate = calculateStartDate(timeRange);

      const url = `${FRED_BASE_URL}/series/observations?series_id=${FRED_FED_RATE_SERIES}&api_key=${apiKey}&observation_start=${formatDate(startDate)}&observation_end=${formatDate(endDate)}&file_type=json`;

      return rateLimiter.call('FRED', async () => {
        const response = await axios.get<FredSeriesResponse>(url);

        // Validate response structure
        if (!response.data?.observations) {
          throw new Error('FRED response missing observations');
        }

        // Transform to NormalizedIndicator
        return normalizeFredData(response.data, timeRange);
      }, RATE_LIMITS.FRED);
    }
    ```

    **Normalization logic (inline or separate utility):**
    - Parse FRED date string (YYYY-MM-DD) to UTC Date
    - Parse value string to number (FRED returns strings)
    - Handle missing values (FRED uses '.' for null) -> use null, not 0
    - Calculate current value (most recent observation)
    - Build historical array

    **Error handling:**
    - Network error: throw with message
    - Missing API key: throw with setup instructions
    - Malformed response: throw with validation error
    - Rate limit exceeded: log warning (client-side can't prevent server 429)

    **Do NOT:**
    - Call API without rate limiter
    - Fill missing data with 0 (use null)
    - Store timestamps as strings (must be Date)
    - Skip response validation
    - Log API key to console (security)
  </action>

    ```typescript
    import { useQuery } from '@tanstack/react-query';
    import { useTransition } from 'react';
    import { getFedRate } from '../api/fred';
    import { downsampleData } from '../utils/downsampling';
    import { TimeRange } from '../types/chart';

    export function useFedRate(timeRange: TimeRange = '1Y') {
      const [isPending, startTransition] = useTransition();

      const query = useQuery({
        queryKey: ['fed-rate', timeRange],
        queryFn: async () => {
          const data = await getFedRate(timeRange);

          // Downsample if too many points
          if (data.historical.length > 365) {
            data.historical = downsampleData(data.historical, 365);
          }

          return data;
        },
        staleTime: 5 * 60 * 1000, // 5 minutes - prevents API exhaustion
        gcTime: 30 * 60 * 1000,   // Keep in cache for 30 min
        refetchInterval: 5 * 60 * 1000, // Auto refetch every 5 min (REAL-04)
        retry: 2,
        refetchOnWindowFocus: false,
      });

      // Use startTransition for non-blocking updates (REAL-05)
      // This prevents UI freeze during data updates

      return {
        ...query,
        isPending: isPending || query.isFetching,
      };
    }
    ```

    **Key configuration:**
    - staleTime: 5 min (matches FRED rate limit cache strategy)
    - refetchInterval: 5 min (minute-level polling for economic data)
    - retry: 2 (handle transient network errors)
    - refetchOnWindowFocus: false (don't waste API calls on tab switch)

    **Do NOT:**
    - Set staleTime < 5 min (exhausts quota)
    - Skip downsampling (chart may freeze)
    - Use useEffect + setInterval (use TanStack Query refetchInterval)
    - Skip useTransition (state updates may freeze UI)
  </action>

    ```typescript
    import { Component, ReactNode } from 'react';

    interface Props {
      children: ReactNode;
      fallback?: ReactNode;
    }

    interface State {
      hasError: boolean;
      error: Error | null;
    }

    export class ErrorBoundary extends Component<Props, State> {
      state = { hasError: false, error: null };

      static getDerivedStateFromError(error: Error): State {
        return { hasError: true, error };
      }

      componentDidCatch(error: Error, errorInfo: any) {
        console.error('ErrorBoundary caught:', error, errorInfo);
      }

      render() {
        if (this.state.hasError) {
          return this.props.fallback || (
            <div className="p-4 bg-red-900/20 rounded text-red-400">
              <p className="font-medium">数据加载失败</p>
              <p className="text-sm">{this.state.error?.message}</p>
            </div>
          );
        }

        return this.props.children;
      }
    }
    ```
