    ```typescript
    // Format large numbers in Chinese convention (万, 亿)
    export function formatChineseNumber(value: number): string {
      if (value >= 100000000) {
        return `${(value / 100000000).toFixed(2)}亿`;
      }
      if (value >= 10000) {
        return `${(value / 10000).toFixed(2)}万`;
      }
      return value.toLocaleString('zh-CN');
    }

    // Format percentage (handle both 0.05 and 5 formats)
    export function formatPercentage(value: number, decimals: number = 2): string {
      // If value is < 1, assume it's already a decimal (0.05 = 5%)
      const percentage = value < 1 ? value * 100 : value;
      return `${percentage.toFixed(decimals)}%`;
    }

    // Format price with unit and Chinese convention
    export function formatPrice(value: number, unit: string): string {
      // For indices, use Chinese format
      if (unit === 'index') {
        return formatChineseNumber(value);
      }

      // For USD, use standard format
      if (unit === 'USD') {
        return `$${value.toLocaleString('zh-CN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;
      }

      // For BTC/ETH, no dollar sign, just number
      if (unit === 'BTC' || unit === 'ETH') {
        return `${value.toLocaleString('zh-CN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })} ${unit}`;
      }

      // Default: number with unit
      return `${value.toLocaleString('zh-CN')} ${unit}`;
    }

    // Format date for chart axis (Chinese labels)
    export function formatChartDate(date: Date, timeRange: TimeRange): string {
      const formats: Record<TimeRange, string> = {
        '1M': 'MM/dd',
        '3M': 'MM/dd',
        '6M': 'yyyy/MM',
        '1Y': 'yyyy/MM',
        'ALL': 'yyyy/MM',
      };
      return format(date, formats[timeRange], { locale: zhCN });
    }
    ```

    **Key improvements:**

    - Use US number separators (,) (use Chinese convention or consistent format)
    - Create ambiguity between 5 and 5% (normalize to percentage first)
    - Skip locale parameter (zhCN ensures Chinese formatting)
  </action>
