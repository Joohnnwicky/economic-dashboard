    **src/api/alphavantage.ts:**
    ```typescript
    import { rateLimiter } from './rate-limiter';
    import { ALPHA_VANTAGE_BASE_URL, RATE_LIMITS } from '../constants/api';
    import { NormalizedIndicator } from '../types/indicator';
    import { parseUTCDate } from '../utils/utc';

    let alphaVantageCallCount = 0;
    const MAX_AV_CALLS_PER_DAY = 25;

    // Get daily time series for an index
    export async function getIndexData(symbol: string): Promise<NormalizedIndicator> {
      const apiKey = import.meta.env.VITE_ALPHA_VANTAGE_API_KEY;

      if (!apiKey) {
        throw new Error('VITE_ALPHA_VANTAGE_API_KEY not set in .env.local');
      }

      alphaVantageCallCount++;
      console.warn(`[Alpha Vantage] Call #${alphaVantageCallCount} of ${MAX_AV_CALLS_PER_DAY} daily quota`);

      if (alphaVantageCallCount >= MAX_AV_CALLS_PER_DAY - 5) {
        console.error(`[Alpha Vantage] WARNING: Approaching daily quota! ${MAX_AV_CALLS_PER_DAY - alphaVantageCallCount} remaining`);
      }

      const url = `${ALPHA_VANTAGE_BASE_URL}?function=TIME_SERIES_DAILY&symbol=${symbol}&apikey=${apiKey}`;

      return rateLimiter.call('AlphaVantage', async () => {
        const response = await axios.get<AlphaVantageDailyResponse>(url);

        if (!response.data?.['Time Series (Daily)']) {
          throw new Error('Alpha Vantage response missing Time Series');
        }

        const timeSeries = response.data['Time Series (Daily)'];

        const historical = Object.entries(timeSeries)
          .map(([dateStr, values]) => ({
            timestamp: parseUTCDate(dateStr),
            value: parseFloat(values['4. close']), // Use close price
          }))
          .sort((a, b) => a.timestamp.getTime() - b.timestamp.getTime()); // Chronological

        const current = historical[historical.length - 1];

        return {
          id: symbol.toLowerCase(),
          name: getIndexName(symbol),
          value: current?.value || 0,
          unit: 'index',
          timestamp: current?.timestamp || new Date(),
          historical: historical.slice(-365), // Last 365 days max
        };
      }, RATE_LIMITS.ALPHA_VANTAGE);
    }

    // Helper to map symbol to name
    function getIndexName(symbol: string): string {
      const names: Record<string, string> = {
        'DJI': '道琼斯指数',
        'NASDAQ': '纳斯达克指数',
        'SPX': '标普500指数',
      };
      return names[symbol] || symbol;
    }
    ```

    **Key considerations:**
    - Alpha Vantage returns string values (parse to number)
    - Use '4. close' price (close is standard for indices)
    - Sort by date (API returns unordered)
    - Limit to 365 days (prevent big data)
    - Track quota and warn when approaching limit

    **Do NOT:**
    - Skip quota tracking
    - Set cache < 60 min
    - Use all data points (limit to 365)
    - Skip date sorting
  </action>

    **Wait - CRITICAL contradiction:**
    - Alpha Vantage: 25 calls/day = ~1 call per hour max
    - Requirement REAL-03: Minute-level updates

    **Resolution:**
    - Can't do true minute updates with 25/day quota
    - Strategy: Fetch once per hour (quota-safe), display cached value
    - Show "Last updated" timestamp clearly

    **src/hooks/useIndices.ts:**
    ```typescript
    import { useQueries } from '@tanstack/react-query';
    import { useTransition } from 'react';
    import { getIndexData } from '../api/alphavantage';
    import { US_INDICES } from '../constants/indicators';

    export function useIndices() {
      const [isPending, startTransition] = useTransition();

      // Fetch all three indices in parallel
      const queries = useQueries({
        queries: [
          {
            queryKey: ['index', 'dow-jones'],
            queryFn: () => getIndexData(US_INDICES.dowJones.symbol),
            staleTime: 60 * 60 * 1000,    // 60 min - quota protection
            gcTime: 2 * 60 * 60 * 1000,   // Keep 2 hours
            // NO refetchInterval - quota can't support minute updates!
            retry: 1,
            refetchOnWindowFocus: false,
          },
          {
            queryKey: ['index', 'nasdaq'],
            queryFn: () => getIndexData(US_INDICES.nasdaq.symbol),
            staleTime: 60 * 60 * 1000,
            gcTime: 2 * 60 * 60 * 1000,
            retry: 1,
            refetchOnWindowFocus: false,
          },
          {
            queryKey: ['index', 'sp500'],
            queryFn: () => getIndexData(US_INDICES.sp500.symbol),
            staleTime: 60 * 60 * 1000,
            gcTime: 2 * 60 * 60 * 1000,
            retry: 1,
            refetchOnWindowFocus: false,
          },
        ],
      });

      return {
        dowJones: queries[0].data,
        nasdaq: queries[1].data,
        sp500: queries[2].data,
        isLoading: queries.some(q => q.isLoading),
        isError: queries.some(q => q.isError),
        errors: queries.map(q => q.error),
        isPending: isPending || queries.some(q => q.isFetching),
      };
    }
    ```

    **CRITICAL Note:**
    - REAL-03 requires minute-level updates, but Alpha Vantage free tier cannot support this
    - With 25/day quota, max refresh is ~1/hour (if only indices, no other calls)
    - Alternative for minute updates: Paid Alpha Vantage ($50/month) or WebSocket sources
    - For Phase 1: Accept hour-level updates with clear "Last updated" timestamp
    - Document limitation in UI: "数据每小时更新一次 (API配额限制)"

    **Do NOT:**
    - Set refetchInterval < 60 min (exhausts quota in minutes)
    - Skip quota warning comments
    - Fetch sequentially (parallel saves time)
  </action>
